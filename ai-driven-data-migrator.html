def transform_account_name(
    name: Optional[str],
) -> str:
    if name is None:
        raise ValueError(
            "Account name is required"
        )
    name = name.strip()
    if len(name) > 100:
        logger.warning(
            f"Truncating from {len(name)}"
            f" to 100 characters"
        )
        return name[:100]
    return name

def transform_doc_number(
    tran_id: Optional[str],
) -> Optional[str]:
    if tran_id is None:
        return None
    if not isinstance(tran_id, str):
        raise ValueError(
            f"Transaction ID must be "
            f"string, got {type(tran_id)}"
        )
    tran_id = tran_id.strip()
    if len(tran_id) > 21:
        logger.debug(
            f"Truncating to 21 chars"
        )
        return tran_id[:21]
    return tran_id
